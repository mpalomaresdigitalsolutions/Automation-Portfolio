9. WAVE 1 MIGRATION SQL (APPLIED TO PRODUCTION)
================================================================

-- WAVE 1: Critical cross-tenant leak fixes
-- Client portal project: dkkriesneublbmrihgvp
-- Transactional + idempotent (safe to re-run)

BEGIN;

-- 1. clients: was readable by ANY authenticated user, now scoped
DROP POLICY IF EXISTS clients_select_authenticated ON public.clients;
CREATE POLICY clients_select_own_or_admin
  ON public.clients
  FOR SELECT
  USING (
    (is_admin() OR id = my_client_id())
    AND deleted_at IS NULL
  );

-- 2. conversations: was readable by ANY authenticated user, now scoped
DROP POLICY IF EXISTS conversations_select ON public.conversations;
CREATE POLICY conversations_select_own_or_admin
  ON public.conversations
  FOR SELECT
  USING (
    (is_admin() OR client_id = my_client_id())
    AND deleted_at IS NULL
  );

-- 3. messages: was readable by ANY authenticated user, now scoped via parent conversation
DROP POLICY IF EXISTS messages_select ON public.messages;
CREATE POLICY messages_select_own_or_admin
  ON public.messages
  FOR SELECT
  USING (
    (
      is_admin()
      OR conversation_id IN (
        SELECT id FROM public.conversations WHERE client_id = my_client_id()
      )
    )
    AND deleted_at IS NULL
  );

-- 4. notifications: was WITH CHECK (true) -- fully open insert
-- Table has no client_id; it has client_email (text). SELECT policy is
-- already admin-only, so this table is an admin-facing alert feed populated
-- by client actions. Scope insert so a client can only create a notification
-- tagged with their own email, not spoof another client's.
DROP POLICY IF EXISTS notifications_insert ON public.notifications;
CREATE POLICY notifications_insert_scoped
  ON public.notifications
  FOR INSERT
  WITH CHECK (
    is_admin()
    OR client_email = auth.email()
  );

-- 5. activity_log: insert had no client scoping, only "authenticated"
DROP POLICY IF EXISTS auth_insert_activity ON public.activity_log;
CREATE POLICY activity_insert_scoped
  ON public.activity_log
  FOR INSERT
  WITH CHECK (
    is_admin()
    OR project_id IN (
      SELECT id FROM public.projects WHERE client_id = my_client_id()
    )
  );

-- 6. contracts: insert had no client scoping, only "authenticated"
DROP POLICY IF EXISTS authenticated_insert ON public.contracts;
CREATE POLICY contracts_insert_scoped
  ON public.contracts
  FOR INSERT
  WITH CHECK (
    is_admin()
    OR client_id = my_client_id()
  );

COMMIT;

-- WAVE 2: Function hardening (fixed search_path), no policy rewrites needed
-- Client portal project: dkkriesneublbmrihgvp
-- Transactional + idempotent

BEGIN;

-- 1. Harden is_admin() in place (30 policies reference this name; keep it stable)
CREATE OR REPLACE FUNCTION public.is_admin()
 RETURNS boolean
 LANGUAGE sql
 STABLE SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
  SELECT EXISTS (
    SELECT 1 FROM public.portal_users
    WHERE email = auth.email()
    AND role = 'admin'
  );
$function$;

-- 2. Harden my_client_id() in place (30 policies reference this name; keep it stable)
CREATE OR REPLACE FUNCTION public.my_client_id()
 RETURNS bigint
 LANGUAGE sql
 STABLE SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
  SELECT client_id FROM public.portal_users
  WHERE email = auth.email()
  LIMIT 1;
$function$;

-- 3. Harden purge_old_soft_deletes() -- logic unchanged, only search_path added
CREATE OR REPLACE FUNCTION public.purge_old_soft_deletes()
 RETURNS void
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
BEGIN
  DELETE FROM public.messages WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.conversations WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.deliverables WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.milestones WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.invoices WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.contracts WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.activity_log WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.projects WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
  DELETE FROM public.clients WHERE deleted_at IS NOT NULL AND deleted_at < now() - interval '7 days';
END;
$function$;

-- 4. Harden soft_delete_row() -- logic unchanged, only search_path added
CREATE OR REPLACE FUNCTION public.soft_delete_row()
 RETURNS trigger
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
BEGIN
  EXECUTE format('UPDATE public.%I SET deleted_at = now() WHERE id = $1', TG_TABLE_NAME)
  USING OLD.id;
  RETURN NULL; -- cancels the actual DELETE
END;
$function$;

-- 5. Restrict execute grants: these two are cron/trigger-only, not client-callable
REVOKE EXECUTE ON FUNCTION public.soft_delete_row() FROM anon, authenticated;
REVOKE EXECUTE ON FUNCTION public.purge_old_soft_deletes() FROM anon, authenticated;

COMMIT;
